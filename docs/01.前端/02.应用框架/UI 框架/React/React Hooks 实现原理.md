核心原理：闭包存储状态、索引维护次序、变更触发渲染

# useState 模拟实现

```js
const globaltate = []
let hookIndex = 0
function useState(initState) {
    const currentIndex = hookIndex++
    const state = globaltate[currentIndex] || initState
    const setState = (_state) => {
        globaltate[currentIndex] = _state
        render() // 触发 react 渲染
    }
   
    return [state, setState]
}
```
